ko00001_htext2df=function(input_file, output_file,header_num = 0) {
    # Read the input file
    input <- readLines(input_file)

    # Open the output file
    output <- file(output_file, "w")

    count <- list()
    while (header_num > 0) {
        input <- input[-1]
        header_num <- header_num - 1
    }

    # Process the input file
    for (line in input) {
        if (grepl("^A", line)) {
            a <- line
            a <- sub(" ", "\t", a)
            next
        }
        if (grepl("^B  ", line)) {
            b <- line
            b <- sub("B  ", "", b)
            b <- sub(" ", "\t", b)
            next
        }
        if (grepl("^C    ", line)) {
            c <- line
            c <- sub("C    ", "", c)
            c <- sub(" ", "\t", c)
            next
        }
        if (grepl("^D      ", line)) {
            d <- line
            d <- sub("D      ", "", d)
            d <- sub("  ", "\t", d)
            writeLines(paste(a, b, c, d, sep = "\t"), output)
        }
    }

    # Close the output file
    close(output)
}
